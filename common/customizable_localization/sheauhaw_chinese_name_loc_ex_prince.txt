
Sheauhaw_GetPrincePrincess = {
	type = character

	text = {
		localization_key = prince_father
		trigger = {
			exists = father
			father = {
				exists = primary_title
				primary_title.tier = tier_kingdom
			}
		}
	}

	text = {
		localization_key = princess_mother
		trigger = {
			exists = mother
			mother = {
				exists = primary_title
				primary_title.tier = tier_kingdom
			}
		}
	}
}
